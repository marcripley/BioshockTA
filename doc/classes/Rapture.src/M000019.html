  def fight_enemy
    puts "Choose your weapon"
    @weapons = @player.weapons
    1.upto(@weapons.size) {|n| puts n.to_s + " - " + @weapons[n-1].to_s}
    puts (@weapons.size + 1).to_s + " - Screw this, just run"
    weapon_input = gets.chomp
    battle(weapon_input)
  end
