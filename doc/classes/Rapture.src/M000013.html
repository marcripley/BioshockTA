  def show_enemies
    chance = find_room_in_rapture(@player.location).enemies
    if chance == 10
      @enemy = Enemy.new("Leadheaded Splicer", 100, rand(10), :tk, :shock)
      puts "There is a " + @enemy.name + " in the room! It's seen you! It runs straight for you..."
      fight_choice
    end
  end
