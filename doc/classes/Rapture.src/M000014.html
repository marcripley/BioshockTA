  def go(direction)
    current_room = @player.location
    @player.location = find_room_in_direction(direction)
    if current_room == @player.location
      puts "\nYou cannot go " + direction.to_s
    else
      puts "\nYou go " + direction.to_s
      show_current_description
      show_current_items
      show_enemies
    end
  end
